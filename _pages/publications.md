<script>
(function(){
  var search = document.getElementById('pubSearch');
  var count = document.getElementById('pubCount');
  var empty = document.getElementById('pubEmpty');
  var buttons = Array.prototype.slice.call(document.querySelectorAll('.pub-filter-btn'));
  var items = Array.prototype.slice.call(document.querySelectorAll('.pub-filter-item'));
  var activeTopic = 'all';

  function applyFilter(){
    var q = (search.value || '').trim().toLowerCase();
    var shown = 0;
    items.forEach(function(item){
      var topicOk = activeTopic === 'all' || item.dataset.topic === activeTopic;
      var textOk = !q || (item.dataset.search || '').toLowerCase().indexOf(q) !== -1;
      var show = topicOk && textOk;
      item.style.display = show ? '' : 'none';
      if(show) shown += 1;
    });
    count.textContent = shown + ' of ' + items.length + ' publications shown';
    empty.style.display = shown ? 'none' : 'block';
  }

  buttons.forEach(function(btn){
    btn.addEventListener('click', function(){
      buttons.forEach(function(b){ b.classList.remove('active'); });
      btn.classList.add('active');
      activeTopic = btn.dataset.topic || 'all';
      applyFilter();
    });
  });
  if(search) search.addEventListener('input', applyFilter);
  applyFilter();
})();
</script>
